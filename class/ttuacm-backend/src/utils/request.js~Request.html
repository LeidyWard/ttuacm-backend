    Request

    Handles the creation of API Calls

    Example:

    const request = new Request('v2', 'auth')

request.post().path('/register')
 .body(newUser)
 .end().then((res) => {
   console.log('Success!')
 }).catch((err) => {
   console.log('Failure!')
   console.error(err)
 })
