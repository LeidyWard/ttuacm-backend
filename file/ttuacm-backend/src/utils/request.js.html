    const axios = require('axios')
const functions = require('firebase-functions')
const ErrorMessages = require('./request.errors')

const { connections } = functions.config()

process.env = connections

// These are the required environment variables
const { protocol, host } = process.env

const AVAILABLE_VERSIONS = new Set([ 'v2' ])
const AVAILABLE_SERVICES= new Set([
  'auth',
  'contacts',
  'events',
  'profile',
  'email',
])

/**
 * Handles the errors returned from call
 *
 * @param {object} err Axios Error
 */
function errorHandler(err) {
  console.error(err)
  if (err.response) {
    // Server responded with a status code outside 2xx range.
    console.error(ErrorMessages.ResponseError(err))
  } else if (err.request) {
    // No response recieved
    console.error(ErrorMessages.NoResponseError())
  } else {
    // Who knows
    console.error(ErrorMessages.UnknownServerError())
  }
}

/**
 * Handles the creation of API Calls
 *
 * @example
 * const request = new Request('v2', 'auth')
 *
 * request.post().path('/register')
 *  .body(newUser)
 *  .end().then((res) => {
 *    console.log('Success!')
 *  }).catch((err) => {
 *    console.log('Failure!')
 *    console.error(err)
 *  })
 */
class Request {
  /**
   * Sets defaults for the connection
   * @param {string} version version of the api to use
   * @param {string} service service of the api to use
   */
  constructor(version, service) {
    if (!AVAILABLE_VERSIONS.has(version)) throw ErrorMessages.BadVersion()
    if (!AVAILABLE_SERVICES.has(service)) throw ErrorMessages.BadService()

    this.version = version
    this.status = 0
    this.path = ''
    this.method = 'get'
    this.config = {
      data: {},
      params: {}
    }

    this.instance = axios.create({
      baseURL: `${protocol}://${host}/api/${version}`,
      timeout: 5000
    })
  }


  /**
   * Sets the body of the request object
   *
   * @param {object} body
   * @returns {Request}
   */
  body(body) {
    Object.keys(body).forEach((key) => {
      this.config.data[key] = body[key];
    });
    return this;
  }

  /**
   * Sets the params of the request object
   *
   * @param {object} params
   * @returns {Request}
   */
  params(params) {
    Object.keys(params).forEach((key) => {
      this.config.params[key] = params[key];
    });
    return this;
  }

  /**
   * Sets the path without the /
   *
   * @param {string} path
   * @returns {Request}
   */
  path(path) {
    this.path = path[0] === '/' ? path : path.substr(1)
    return this
  }

  /**
   * Sets the method for the call
   *
   * @private
   * @param {string} method
   * @returns {Request}
   */
  _setMethod(method) {
    this.method = method
    return this
  }

  post() { return this._setMethod('post') }

  del() { return this._setMethod('delete') }

  put() { return this._setMethod('put') }

  patch() { return this._setMethod('patch') }

  get() { return this._setMethod('get') }

  /**
   * Composes the call altogether
   * @returns {Promise.<Response, Error>}
   */
  end() {
    return new Promise(async (resolve, reject) => {
      this.config.method = this.method
      this.config.url = `/${this.path}`

      try {
        const res = this.instance(this.config)
        resolve(res)
      } catch (err) {
        errorHandler(err)
        reject(err)
      }

    })
  }
}

module.exports = Request
